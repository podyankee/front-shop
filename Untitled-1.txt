'use strict';

var smartgrid = require('smart-grid');


module.exports = function() {
	function grid (done) {
	delete require.cashe[require.resolve($.config.smartgridConfig)];
	let gridOptions = require($.config.smartgridConfig);
	smartgrid('./source/style/layout', gridOptions);
	done();
	return $.gulp.dest('./source/style/helpers');
};
};
